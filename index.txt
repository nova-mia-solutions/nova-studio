<script>
const saved = localStorage.getItem('nova_key');
if(saved) document.getElementById('k').value = saved;

async function go(){
  const k = document.getElementById('k').value.trim();
  const btn = document.getElementById('btn');
  if(!k){ show('Please enter your Anthropic API key','err'); return; }
  btn.textContent='⚙️ Verifying…'; btn.disabled=true;
  show('Checking API key…','ok');
  try {
    const r = await fetch('https://api.anthropic.com/v1/messages',{
      method:'POST',
      headers:{'Content-Type':'application/json','x-api-key':k,'anthropic-version':'2023-06-01'},
      body:JSON.stringify({model:'claude-haiku-4-5-20251001',max_tokens:5,messages:[{role:'user',content:'hi'}]})
    });
    const d = await r.json();
    if(d.content){
      localStorage.setItem('nova_key',k);
      show('✅ Verified! NOVA loading…','ok');
      setTimeout(()=>{
        document.body.innerHTML = `
        <div style="font-family:Inter,system-ui,sans-serif;background:#0d1117;min-height:100vh;display:flex;align-items:center;justify-content:center;color:#e2e8f0;">
          <div style="text-align:center">
            <div style="font-size:48px;font-weight:900;background:linear-gradient(135deg,#6366f1,#a855f7);-webkit-background-clip:text;-webkit-text-fill-color:transparent;margin-bottom:16px">N</div>
            <h2 style="font-size:20px;font-weight:700;color:#f1f5f9;margin-bottom:8px">NOVA AI Studio</h2>
            <p style="color:#10b981;font-size:14px;margin-bottom:4px">✅ Connected & Ready</p>
            <p style="color:#6b8aad;font-size:12px">Full dashboard coming in next update</p>
          </div>
        </div>`;
      },1200);
    } else {
      show('Invalid key — '+(d.error?.message||'check and retry'),'err');
      btn.textContent='⚡ Launch NOVA'; btn.disabled=false;
    }
  } catch(e){
    show('Error: '+e.message,'err');
    btn.textContent='⚡ Launch NOVA'; btn.disabled=false;
  }
}

function show(msg,cls){
  const el=document.getElementById('st');
  el.textContent=msg; el.className='status '+cls; el.style.display='block';
}

document.getElementById('k').addEventListener('keydown',e=>{ if(e.key==='Enter') go(); });
